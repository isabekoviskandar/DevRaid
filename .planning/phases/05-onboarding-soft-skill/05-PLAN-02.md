**State Management:**
```typescript
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { QUESTIONS } from '@/lib/onboarding-questions'
import { calculateHexagon } from '@/lib/skill-calculator'
import { SoftSkillHexagon } from '@/types'

const TOTAL_QUESTIONS = 20

export default function OnboardingPage() {
  const router = useRouter()

  const [currentStep, setCurrentStep] = useState(0)  // 0-indexed, 0 = question 1
  const [answers, setAnswers] = useState<Record<string, 'A'|'B'|'C'|'D'>>({})
  const [hexagon, setHexagon] = useState<SoftSkillHexagon | null>(null)
  const [isSubmitting, setIsSubmitting] = useState(false)

  const isLastStep = currentStep === TOTAL_QUESTIONS - 1
  const currentQuestion = QUESTIONS[currentStep]
  const hasAnswered = answers[currentQuestion?.id] !== undefined
  const progressPercent = ((currentStep + 1) / TOTAL_QUESTIONS) * 100
```

**Handler Functions:**
```typescript
const handleSelectAnswer = (value: 'A'|'B'|'C'|'D') => {
  setAnswers(prev => ({
    ...prev,
    [currentQuestion.id]: value
  }))
}

const handleNext = async () => {
  if (!hasAnswered) return

  if (isLastStep) {
    // Step 20 → Calculate hexagon
    setIsSubmitting(true)
    const result = calculateHexagon(answers)
    setHexagon(result)

    // TODO: POST to backend (Phase 6+)
    // Optional: localStorage backup
    localStorage.setItem('soft_skills_completed', JSON.stringify(result))

    setIsSubmitting(false)
  } else {
    // Normal step transition
    setCurrentStep(prev => prev + 1)
  }
}

const handleGoToProfile = () => {
  // Frontend trick: Assume middleware will redirect on GET /hero/me
  // Set flag in localStorage so page recognizes completion
  localStorage.setItem('onboarding_completed', 'true')
  router.push('/hero/me')
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'

// Routes that require authentication
const PROTECTED_ROUTES = ['/hero', '/raids', '/captain']

// Routes that require completed onboarding
const ONBOARDING_REQUIRED = ['/hero/me', '/hero', '/raids', '/captain']

// Routes that are always public
const PUBLIC_ROUTES = ['/login', '/register', '/onboarding']

export async function middleware(request: NextRequest) {
  const pathname = request.nextUrl.pathname

  // Skip API routes, static files, etc
  if (pathname.startsWith('/api') || pathname.startsWith('/_next')) {
    return NextResponse.next()
  }

  // Check if route requires authentication
  const isProtected = PROTECTED_ROUTES.some(route => pathname.startsWith(route))
  const isPublic = PUBLIC_ROUTES.some(route => pathname === route || pathname.startsWith(route))

  // Get token from cookies (adjust if using different storage)
  const token = request.cookies.get('auth_token')?.value

  // If no token and not public route → redirect to login
  if (!token && !isPublic && isProtected) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  // If token exists, check onboarding status
  if (token && ONBOARDING_REQUIRED.some(route => pathname.startsWith(route))) {
    // TODO: Fetch user from backend to check onboarding_completed flag
    // For now, check localStorage (client-side hint)
    // Production: decode JWT or make API call to verify

    // Stub: Assume middleware can't easily access user state
    // Actual check happens on client-side, or via API route wrapper
    // For Phase 5, we'll rely on:
    // 1. localStorage flag set by page.tsx after completion
    // 2. Optional: next-auth middleware integration (Phase 6+)
  }

  return NextResponse.next()
}

export const config = {
  matcher: [
    // Match all routes except those starting with:
    // - _next (internal Next.js files)
    // - api (API routes, handled separately)
    // - (public assets like images, fonts)
    '/((?!_next|api|.*\\..*|static|public).*)',
  ],
}
```
