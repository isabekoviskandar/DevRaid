Add this event handler inside the component (before the return):

```typescript
function handleSkillKeyDown(e: React.KeyboardEvent<HTMLInputElement>) {
  if (e.key === 'Enter' || e.key === ',') {
    e.preventDefault()
    const trimmed = skillInput.trim().replace(/,$/, '')
    if (trimmed && !draft.hard_skills.includes(trimmed)) {
      setDraft(d => ({ ...d, hard_skills: [...d.hard_skills, trimmed] }))
    }
    setSkillInput('')
  }
}

async function handleSave() {
  setSaveState('saving')
  const token = authStore.getToken()
  if (token !== null) {
    await api.auth.patchUser(token, draft)
  }
  authStore.updateUser(draft)
  const updated = authStore.getUser()
  if (updated !== null) {
    setUser(updated)
  }
  setSaveState('saved')
  setTimeout(() => {
    setSaveState('idle')
    setIsEditing(false)
  }, 1500)
}
```
